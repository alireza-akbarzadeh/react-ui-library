## Scroll to element with function and re-render

<Playground>
  <ThemeWrapper>
    <StateWrapper initialState={0}>
      {({ state: count, setState: setCount }) => (
        <ListRegistryProvider>
          <ListRegistryConsumer>
            {({ getItem }) => (
              <React.Fragment>
                 <Looper start={0} end={40} step={10} render={i => (
                  <Rhythm key={i} m={1}>
                    <Button color="primary" onClick={() => { setCount(count + 1); getItem(i).scrollIntoView({ behavior: 'smooth', block: 'nearest', inline: 'start' })}}>{`Scroll to ${i}`}</Button>
                  </Rhythm>
                )} />
                <div style={{marginTop: '20px', height: '100px', overflow: 'auto'}}>
                  <Looper start={0} end={40} render={i => (
                    <ListRegistryItem id={i} key={i}>
                      {({ ref }) => (
                        <div ref={ref}>
                          <Rhythm px={2} my={1}>{`Hello world ${i}${count > 0 ? ` ${count}` : ''}`}</Rhythm>
                        </div>
                      )}
                    </ListRegistryItem>
                  )} />
                </div>
              </React.Fragment>
            )}
          </ListRegistryConsumer>
        </ListRegistryProvider>
      )}
    </StateWrapper>
  </ThemeWrapper>
</Playground>
