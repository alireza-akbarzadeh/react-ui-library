## Full context

<Playground>
  <ThemeWrapper>
    <StateWrapper>
      {({ state: variant, setState: setVariant }) => (
        <Toasts position="top-right" variant={variant}>
          <ToastConsumer>
            {({ createNotification, clearNotifications }) => (
              <Rhythm m={2}>
                <Button
                  color="primary"
                  onClick={() =>
                    createNotification(
                      <Toast level="info" title="Hello world">
                        This is an info toast.
                      </Toast>,
                    )
                  }
                >
                  Info toast
                </Button>
                <Button
                  color="success"
                  onClick={() =>
                    createNotification(
                      <Toast id="demo-success" level="success" title="Hello world">
                        This is an overwriting success toast.
                        <div
                          style={{ position: 'absolute', bottom: '4px', right: '8px', fontSize: '10px', opacity: 0.4 }}
                        >
                          {Date.now()}
                        </div>
                      </Toast>,
                    )
                  }
                >
                  Success toast
                </Button>
                <Button
                  color="warning"
                  onClick={() =>
                    createNotification(
                      <Toast level="warning" title="Hello world" permanent>
                        This is a permanent warning toast.
                      </Toast>,
                    )
                  }
                >
                  Warning toast
                </Button>
                <Button
                  color="danger"
                  onClick={() =>
                    createNotification(
                      <Toast level="danger" title="Hello world">
                        This is a danger toast.
                      </Toast>,
                    )
                  }
                >
                  Danger toast
                </Button>
                <StyledButton
                  primaryColor="#556270"
                  inverseColor="#fff"
                  hoverPrimaryColor="#393945"
                  activePrimaryColor="#708194"
                  onClick={() =>
                    createNotification(
                      <StyledToast title="Hello world" levelColor="#556270" levelInverseColor="#fff">
                        This is a styled toast.
                      </StyledToast>,
                    )
                  }
                >
                  Styled toast
                </StyledButton>
                <Button color="neutral" weight="outline" onClick={() => clearNotifications()}>
                  Clear all
                </Button>
                <Button color="neutral" onClick={() => setVariant(variant === 'colored' ? undefined : 'colored')}>
                  Toggle variant
                </Button>
              </Rhythm>
            )}
          </ToastConsumer>
        </Toasts>
      )}
    </StateWrapper>
  </ThemeWrapper>
</Playground>
