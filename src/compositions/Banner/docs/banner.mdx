## Full context

<Playground>
  <ThemeWrapper>
    <Banners style={{ marginTop: '18px' }}>
      <BannerConsumer>
        {({ createNotification, clearNotifications }) => (
          <Rhythm m={2}>
            <Button
              color="primary"
              onClick={() =>
                createNotification(
                  <Banner level="info">
                    This is an info banner because something interesting happened. Exciting!
                  </Banner>,
                )
              }
            >
              Info banner
            </Button>
            <Button
              color="danger"
              onClick={() =>
                createNotification(
                  <Banner level="danger">This is a danger banner because something bad happened. Uh oh!</Banner>,
                )
              }
            >
              Danger banner
            </Button>
            <Button
              color="warning"
              onClick={() =>
                createNotification(
                  <Banner level="warning" permanent>
                    This is a permanent warning banner because something not so great happened. Meh.
                  </Banner>,
                )
              }
            >
              Warning banner
            </Button>
            <Button
              color="success"
              onClick={() =>
                createNotification(
                  <Banner id="demo-success" level="success">
                    This is an overwriting success banner because something excellent happened. Yay!
                    <div style={{ position: 'absolute', right: '8px', fontSize: '10px', opacity: 0.4 }}>
                      {Date.now()}
                    </div>
                  </Banner>,
                )
              }
            >
              Success banner
            </Button>
            <Button color="neutral" onClick={() => clearNotifications()}>
              Clear all
            </Button>
          </Rhythm>
        )}
      </BannerConsumer>
    </Banners>
  </ThemeWrapper>
</Playground>
